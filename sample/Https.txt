<VirtualHost *:80>
        <Directory #DIR#>
                Options Indexes FollowSymLinks MultiViews
                AllowOverride All
                Require all granted
        </Directory>
	ServerName #DOMAIN#
        Redirect permanent / https://#DOMAIN#
	DocumentRoot #ROOT#
</VirtualHost>

<IfModule mod_ssl.c>
    <VirtualHost _default_:443>
        <Directory #DIR#>
                Options Indexes FollowSymLinks MultiViews
                AllowOverride All
                Require all granted
        </Directory>
        ServerName #DOMAIN#
        DocumentRoot #ROOT#
        SSLEngine on
        SSLCertificateFile #SSLFILE#
        SSLCertificateKeyFile #SSLKEY#
    </VirtualHost>
</IfModule>
